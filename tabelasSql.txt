create schema VanSport

create table VanSport.Motorista(
 idMotorista int primary key,
 cpf varchar(11) not null,
 email varchar(50) not null,
 senha varchar(10) not null,
 cidade varchar(50) not null,
 nome varchar(50) not null,
 numeroCelular varchar(11) not null,
 foto varchar(100),
 cep int not null,
 numero int not null,
 sexo char not null
)

create table VanSport.Veiculo(
 idVeiculo int primary key,
 placa varchar(7) not null,
 modelo varchar(50) not null,
 marca varchar(20),
 cor varchar(20),
 anoFabricacao int not null,
 capacidade int not null
)

create table VanSport.MotoristaVeiculo(
 idMororistaVeiculo int primary key,
 foreign key(idMotorista) references VanSport.Motorista(idMotorista),
 foreign key(idVeiculo) references VanSport.Veiculo(idVeiculo)
)

create table VanSport.Cliente(
 idCliente int primary key,
 nomeResponsavel varchar(50) not null,
 nomeAluno varchar(50) not null,
 email varchar(50) not null,
 senha varchar(10) not null,
 cidade varchar(50) not null,
 cep int not null,
 numero int not null,
 foto varchar(100),
)

create table VanSport.Aluno(
 idAluno int primary key,
 nome varchar(50) not null,
 cidade varchar(50) not null,
 cep int not null,
 numero int not null
)

create table VanSport.ClienteAluno(
 idClienteAluno int primary key,
 foreign key(idCliente) references VanSport.Cliente(idCliente),
 foreign key(idAluno) references VanSport.Aluno(idAluno)
)

create table VanSport.Turno(
 idTurno int primary key,
 nomeTurno varchar(15)
)

create table VanSport.AlunoTurno(
 idAlunoTurno int primary key,
 foreign key(idAluno) references VanSport.Aluno(idAluno),
 foreign key(idTurno) references VanSport.Turno(idTurno)
)

create table VanSport.Avaliacao(
 idAvaliacao int primary key,
 comentario varchar(100),
 estrela int not null,
 comentarioData date,
 foreign key(idCliente) references VanSport.Cliente(idCliente),
 foreign key(idMotorista) references VanSport.Motorista(idMotorista)
)

create table VanSport.Aviso(
 idAviso int primary key,
 texto varchar(500) not null,
 foto varchar(100),
 comentario varchar(100),
 foreign key(idMotoristaClienteAviso) references VanSport.MotoristaClienteAviso(idMotoristaClienteAviso)
)

create table VanSport.MotoristaCliente(
 idMotoristaCliente int primary key,
 foreign key(idCliente) references VanSport.Cliente(idCliente),
 foreign key(idMotorista) references VanSport.Motorista(idMotorista)
)

create table VanSport.MotoristaClienteAviso(
 idMotoristaClienteAviso int primary key,
 foreign key(idCliente) references VanSport.Cliente(idCliente),
 foreign key(idMotorista) references VanSport.Motorista(idMotorista),
 foreign key(idAviso) references VanSport.Aviso(idAviso)
)

create table VanSport.Chat(
 idChat int primary key,
 foreign key(idCliente) references VanSport.Cliente(idCliente),
 foreign key(idMotorista) references VanSport.Motorista(idMotorista)
)

create table VanSport.Mensagem(
 idMensagem int primary key,
 mensagem text,
 mensagemData datetime
)

create table VanSport.ChatMensagem(
 idChatMensagem int primary key,
 foreign key(idChat) references VanSport.Chat(idChat),
 foreign key(idMensagem) references VanSport.Mensagem(idMensagem)
)
